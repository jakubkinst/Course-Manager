 91:     private function pattern($matches)
 92:     {
 93:         list($mWord) = $matches;
 94:         //    [0] => lllloooonnnnggggwwwoorrdddd
 95: 
 96:         $chars = array();
 97:         preg_match_all(
 98:             '#['.TEXY_MARK.']+|.#u',
 99:             $mWord,
100:             $chars
101:         );
102: 
103:         $chars = $chars[0];
104:         if (count($chars) < $this->wordLimit) return $mWord;
105: 
106:         $consonants = $this->consonants;
107:         $vowels = $this->vowels;
108:         $before_r = $this->before_r;
109:         $before_l = $this->before_l;
110:         $before_h = $this->before_h;
111:         $doubleVowels = $this->doubleVowels;
112: 
113:         $s = array();
114:         $trans = array();
115: 
116:         $s[] = '';
117:         $trans[] = -1;
118:         foreach ($chars as $key => $char) {
119:             if (ord($char{0}) < 32) continue;
120:             $s[] = $char;
121:             $trans[] = $key;
122:         }
123:         $s[] = '';
124:         $len = count($s) - 2;
125: 
126:         $positions = array();
127:         $a = 0; $last = 1;
128: 
129:         while (++$a < $len) {
130:             $hyphen = self::DONT; // Do not hyphenate
131:             do {
132:                 if ($s[$a] === "\xC2\xA0") { $a++; continue 2; } // here and after never
133: 
134:                 if ($s[$a] === '.') { $hyphen = self::HERE; break; }
135: 
136:                 if (isset($consonants[$s[$a]])) {  // consonants
137: 
138:                     if (isset($vowels[$s[$a+1]])) {
139:                         if (isset($vowels[$s[$a-1]])) $hyphen = self::HERE;
140:                         break;
141:                     }
142: 
143:                     if (($s[$a] === 's') && ($s[$a-1] === 'n') && isset($consonants[$s[$a+1]])) { $hyphen = self::AFTER; break; }
144: 
145:                     if (isset($consonants[$s[$a+1]]) && isset($vowels[$s[$a-1]])) {
146:                         if ($s[$a+1] === 'r') {
147:                             $hyphen = isset($before_r[$s[$a]]) ? self::HERE : self::AFTER;
148:                             break;
149:                         }
150: 
151:                         if ($s[$a+1] === 'l') {
152:                             $hyphen = isset($before_l[$s[$a]]) ? self::HERE : self::AFTER;
153:                             break;
154:                         }
155: 
156:                         if ($s[$a+1] === 'h') { // CH
157:                             $hyphen = isset($before_h[$s[$a]]) ? self::DONT : self::AFTER;
158:                             break;
159:                         }
160: 
161:                         $hyphen = self::AFTER;
162:                         break;
163:                     }
164: 
165:                     break;
166:                 }   // end of consonants
167: 
168:                 if (($s[$a] === 'u') && isset($doubleVowels[$s[$a-1]])) { $hyphen = self::AFTER; break; }
169:                 if (isset($vowels[$s[$a]]) && isset($vowels[$s[$a-1]])) { $hyphen = self::HERE; break; }
170: 
171:             } while(0);
172: 
173:             if ($hyphen === self::DONT && ($a - $last > $this->wordLimit*0.6)) $positions[] = $last = $a-1; // Hyphenate here
174:             if ($hyphen === self::HERE) $positions[] = $last = $a-1; // Hyphenate here
175:             if ($hyphen === self::AFTER) { $positions[] = $last = $a; $a++; } // Hyphenate after
176: 
177:         } // while
178:

180:         $a = end($positions);
