 23:     public function __construct($texy)
 24:     {
 25:         $this->texy = $texy;
 26: 
 27:         //$texy->allowed['blocks'] = TRUE;
 28:         $texy->allowed['block/default'] = TRUE;
 29:         $texy->allowed['block/pre'] = TRUE;
 30:         $texy->allowed['block/code'] = TRUE;
 31:         $texy->allowed['block/html'] = TRUE;
 32:         $texy->allowed['block/text'] = TRUE;
 33:         $texy->allowed['block/texysource'] = TRUE;
 34:         $texy->allowed['block/comment'] = TRUE;
 35:         $texy->allowed['block/div'] = TRUE;
 36: 
 37:         $texy->addHandler('block', array($this, 'solve'));
 38:         $texy->addHandler('beforeBlockParse', array($this, 'beforeBlockParse'));
 39: 
 40:         $texy->registerBlockPattern(
 41:             array($this, 'pattern'),
 42:             '#^/--++ *+(.*)'.TEXY_MODIFIER_H.'?$((?:\n(?0)|\n.*+)*)(?:\n\\\\--.*$|\z)#mUi',
 43:             'blocks'
 44:         );
 45:     }

 49:     /**
 50:      * Single block pre-processing.
 51:      * @param  TexyBlockParser
 52:      * @param  string
 53:      * @return void
 54:      */
 55:     public function beforeBlockParse($parser, & $text)
 56:     {
 57:         // autoclose exclusive blocks
 58:         $text = preg_replace(
 59:             '#^(/--++ *+(?!div|texysource).*)$((?:\n.*+)*?)(?:\n\\\\--.*$|(?=(\n/--.*$)))#mi',
 60:             "\$1\$2\n\\--",
 61:             $text
 62:         );
 63:     }

 66: 
 67:     /**
 68:      * Callback for:.
 69:      *   /-----code html .(title)[class]{style}
 70:      *     ....
 71:      *     ....
 72:      *   \----
 73:      *
 74:      * @param  TexyBlockParser
 75:      * @param  array      regexp matches
 76:      * @param  string     pattern name
 77:      * @return TexyHtml|string|FALSE
 78:      */
 79:     public function pattern($parser, $matches)
 80:     {
 81:         list(, $mParam, $mMod, $mContent) = $matches;
 82:         //    [1] => code | text | ...
 83:         //    [2] => ... additional parameters
 84:         //    [3] => .(title)[class]{style}<>
 85:         //    [4] => ... content
 86: 
 87:         $mod = new TexyModifier($mMod);
 88:         $parts = preg_split('#\s+#u', $mParam, 2);
 89:         $blocktype = empty($parts[0]) ? 'block/default' : 'block/' . $parts[0];
 90:         $param = empty($parts[1]) ? NULL : $parts[1];
 91: 
 92:         return $this->texy->invokeAroundHandlers('block', $parser, array($blocktype, $mContent, $param, $mod));
 93:     }

 98:     function outdent($s)
 99:     {

106:     /**
107:      * Finish invocation.
108:      *
109:      * @param  TexyHandlerInvocation  handler invocation
110:      * @param  string   blocktype
111:      * @param  string   content
112:      * @param  string   additional parameter
113:      * @param  TexyModifier
114:      * @return TexyHtml|string|FALSE
115:      */
116:     public function solve($invocation, $blocktype, $s, $param, $mod)
117:     {
118:         $tx = $this->texy;
119:         $parser = $invocation->parser;
120: 
121:         if ($blocktype === 'block/texy') {
122:             $el = TexyHtml::el();
123:             $el->parseBlock($tx, $s, $parser->isIndented());
124:             return $el;
125:         }
126: 
127:         if (empty($tx->allowed[$blocktype])) return FALSE;
128: 
129:         if ($blocktype === 'block/texysource') {
130:             $s = Texy::outdent($s);
131:             if ($s==='') return "\n";
132:             $el = TexyHtml::el();
133:             if ($param === 'line') $el->parseLine($tx, $s);
134:             else $el->parseBlock($tx, $s);
135:             $s = $el->toHtml($tx);
136:             $blocktype = 'block/code'; $param = 'html'; // to be continue (as block/code)
137:         }
138: 
139:         if ($blocktype === 'block/code') {
140:             $s = Texy::outdent($s);
141:             if ($s==='') return "\n";
142:             $s = Texy::escapeHtml($s);
143:             $s = $tx->protect($s, Texy::CONTENT_BLOCK);
144:             $el = TexyHtml::el('pre');
145:             $mod->decorate($tx, $el);
146:             $el->attrs['class'][] = $param; // lang
147:             $el->create('code', $s);
148:             return $el;
149:         }
150: 
151:         if ($blocktype === 'block/default') {
152:             $s = Texy::outdent($s);
153:             if ($s==='') return "\n";
154:             $el = TexyHtml::el('pre');
155:             $mod->decorate($tx, $el);
156:             $el->attrs['class'][] = $param; // lang
157:             $s = Texy::escapeHtml($s);
158:             $s = $tx->protect($s, Texy::CONTENT_BLOCK);
159:             $el->setText($s);
160:             return $el;
161:         }
162: 
163:         if ($blocktype === 'block/pre') {
164:             $s = Texy::outdent($s);
165:             if ($s==='') return "\n";
166:             $el = TexyHtml::el('pre');
167:             $mod->decorate($tx, $el);
168:             $lineParser = new TexyLineParser($tx, $el);
169:             // special mode - parse only html tags
170:             $tmp = $lineParser->patterns;
171:             $lineParser->patterns = array();
172:             if (isset($tmp['html/tag'])) $lineParser->patterns['html/tag'] = $tmp['html/tag'];
173:             if (isset($tmp['html/comment'])) $lineParser->patterns['html/comment'] = $tmp['html/comment'];
174:             unset($tmp);
175: 
176:             $lineParser->parse($s);
177:             $s = $el->getText();
178:             $s = Texy::unescapeHtml($s);
179:             $s = Texy::escapeHtml($s);
180:             $s = $tx->unprotect($s);
181:             $s = $tx->protect($s, Texy::CONTENT_BLOCK);
182:             $el->setText($s);
183:             return $el;
184:         }
185: 
186:         if ($blocktype === 'block/html') {
187:             $s = trim($s, "\n");
188:             if ($s==='') return "\n";
189:             $el = TexyHtml::el();
190:             $lineParser = new TexyLineParser($tx, $el);
191:             // special mode - parse only html tags
192:             $tmp = $lineParser->patterns;
193:             $lineParser->patterns = array();
194:             if (isset($tmp['html/tag'])) $lineParser->patterns['html/tag'] = $tmp['html/tag'];
195:             if (isset($tmp['html/comment'])) $lineParser->patterns['html/comment'] = $tmp['html/comment'];
196:             unset($tmp);
197: 
198:             $lineParser->parse($s);
199:             $s = $el->getText();
200:             $s = Texy::unescapeHtml($s);
201:             $s = Texy::escapeHtml($s);
202:             $s = $tx->unprotect($s);
203:             return $tx->protect($s, Texy::CONTENT_BLOCK) . "\n";
204:         }
205: 
206:         if ($blocktype === 'block/text') {
207:             $s = trim($s, "\n");
208:             if ($s==='') return "\n";
209:             $s = Texy::escapeHtml($s);
210:             $s = str_replace("\n", TexyHtml::el('br')->startTag() , $s); // nl2br
211:             return $tx->protect($s, Texy::CONTENT_BLOCK) . "\n";
212:         }
213: 
214:         if ($blocktype === 'block/comment') {
215:             return "\n";
216:         }
217: 
218:         if ($blocktype === 'block/div') {
219:             $s = Texy::outdent($s);
220:             if ($s==='') return "\n";
221:             $el = TexyHtml::el('div');
222:             $mod->decorate($tx, $el);
223:             $el->parseBlock($tx, $s, $parser->isIndented()); // TODO: INDENT or NORMAL ?
224:             return $el;
225:         }
226: 
227:         return FALSE;
228:     }
