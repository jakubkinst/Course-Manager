257:             $link = $this->getReference($dest);
258: 
259:         // [* image *]
260:         } elseif (strlen($dest)>1 && $dest{0} === '[' && $dest{1} === '*') {
261:             $type = TexyLink::IMAGE;
262:             $dest = trim(substr($dest, 2, -2));
263:             $image = $tx->imageModule->getReference($dest);
264:             if ($image) {
265:                 $link = new TexyLink($image->linkedURL === NULL ? $image->URL : $image->linkedURL);
266:                 $link->modifier = $image->modifier;
267:             }
268:         }
269: 
270:         if (empty($link)) {
271:             $link = new TexyLink(trim($dest));
272:             $this->checkLink($link);
273:         }
274: 
275:         if (strpos($link->URL, '%s') !== FALSE) {
276:             $link->URL = str_replace('%s', urlencode($tx->stringToText($label)), $link->URL);

305:             $popup = isset($link->modifier->classes['popup']);
306:             unset($link->modifier->classes['nofollow'], $link->modifier->classes['popup']);

308:             $link->modifier->decorate($tx, $el);
309:         }
310: 
311:         if ($link->type === TexyLink::IMAGE) {
312:             // image
313:             $el->attrs['href'] = Texy::prependRoot($link->URL, $tx->imageModule->linkedRoot);
314:             $el->attrs['onclick'] = $this->imageOnClick;
315: 
316:         } else {
317:             $el->attrs['href'] = Texy::prependRoot($link->URL, $this->root);
318: 
319:             // rel="nofollow"
320:             if ($nofollow || ($this->forceNoFollow && strpos($el->attrs['href'], '//') !== FALSE))
321:                 $el->attrs['rel'] = 'nofollow';
322:         }

324:         // popup on click

367:     /**
368:      * Checks and corrects $URL.
369:      * @param  TexyLink
370:      * @return void
371:      */
372:     private function checkLink($link)
373:     {
374:         // remove soft hyphens; if not removed by Texy::process()
375:         $link->URL = str_replace("\xC2\xAD", '', $link->URL);
376: 
377:         if (strncasecmp($link->URL, 'www.', 4) === 0) {
378:             // special supported case
379:             $link->URL = 'http://' . $link->URL;
380: 
381:         } elseif (preg_match('#'.TEXY_EMAIL.'$#Au', $link->URL)) {
382:             // email
383:             $link->URL = 'mailto:' . $link->URL;
384: 
385:         } elseif (!$this->texy->checkURL($link->URL, Texy::FILTER_ANCHOR)) {
386:             $link->URL = NULL;
387: 
388:         } else {
389:             $link->URL = str_replace('&amp;', '&', $link->URL); // replace unwanted &amp;
390:         }
391:     }

395:     /**
396:      * Returns textual representation of URL.
397:      * @param  TexyLink
398:      * @return string
399:      */
400:     private function textualUrl($link)
401:     {
402:         if ($this->texy->obfuscateEmail && preg_match('#^'.TEXY_EMAIL.'$#u', $link->raw)) { // email
403:             return str_replace('@', "&#64;<!---->", $link->raw);
404:         }
405: 
406:         if ($this->shorten && preg_match('#^(https?://|ftp://|www\.|/)#i', $link->raw)) {
407: 
408:             $raw = strncasecmp($link->raw, 'www.', 4) === 0 ? 'none://' . $link->raw : $link->raw;
409: 
410:             // parse_url() in PHP damages UTF-8 - use regular expression
411:             if (!preg_match('~^(?:(?P<scheme>[a-z]+):)?(?://(?P<host>[^/?#]+))?(?P<path>(?:/|^)(?!/)[^?#]*)?(?:\?(?P<query>[^#]*))?(?:#(?P<fragment>.*))?()$~', $raw, $parts)) {
412:                 return $link->raw;
413:             }
414: 
415:             $res = '';
416:             if ($parts['scheme'] !== '' && $parts['scheme'] !== 'none')
417:                 $res .= $parts['scheme'] . '://';
418: 
419:             if ($parts['host']  !== '')
420:                 $res .= $parts['host'];
421: 
422:             if ($parts['path']  !== '')
423:                 $res .=  (iconv_strlen($parts['path'], 'UTF-8') > 16 ? ("/\xe2\x80\xa6" . iconv_substr($parts['path'], -12, 12, 'UTF-8')) : $parts['path']);
424: 
425:             if ($parts['query']  !== '') {
426:                 $res .= iconv_strlen($parts['query'], 'UTF-8') > 4 ? "?\xe2\x80\xa6" : ('?' . $parts['query']);
427:             } elseif ($parts['fragment']  !== '') {
428:                 $res .= iconv_strlen($parts['fragment'], 'UTF-8') > 4 ? "#\xe2\x80\xa6" : ('#' . $parts['fragment']);
429:             }
430:             return $res;
431:         }
432: 
433:         return $link->raw;
434:     }
