66:     public function extract($file)
67:     {
68:         $pInfo = pathinfo($file);
69:         $data = array();
70:         $tokens = token_get_all(file_get_contents($file));
71:         $next = false;
72:         foreach ($tokens as $c)
73:         {
74:             if(is_array($c)) {
75:                 if ($c[0] != T_STRING && $c[0] != T_CONSTANT_ENCAPSED_STRING) continue;
76:                 if ($c[0] == T_STRING && isset($this->functions[$c[1]])) {
77:                     $next = $this->functions[$c[1]];
78:                     continue;
79:                 }
80:                 if ($c[0] == T_CONSTANT_ENCAPSED_STRING && $next == 1) {
81:                     $data[substr($c[1], 1, -1)][] = $pInfo['basename'] . ':' . $c[2];
82:                     $next = false; 
83:                 }
84:             } else {
85:                 if ($c == ')') $next = false;
86:                 if ($c == ',' && $next != false) $next -= 1;
87:             }
88:         }
89:         return $data;
90:     }
