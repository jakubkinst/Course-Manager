216:     /**
217:      * Callback for: **.... .(title)[class]{style}**:LINK.
218:      *
219:      * @param  TexyLineParser
220:      * @param  array      regexp matches
221:      * @param  string     pattern name
222:      * @return TexyHtml|string|FALSE
223:      */
224:     public function patternPhrase($parser, $matches, $phrase)
225:     {
226:         list(, $mContent, $mMod, $mLink) = $matches;
227:         //    [1] => **
228:         //    [2] => ...
229:         //    [3] => .(title)[class]{style}
230:         //    [4] => LINK
231: 
232:         $tx = $this->texy;
233:         $mod = new TexyModifier($mMod);
234:         $link = NULL;
235: 
236:         $parser->again = $phrase !== 'phrase/code' && $phrase !== 'phrase/quicklink';
237: 
238:         if ($phrase === 'phrase/span' || $phrase === 'phrase/span-alt') {
239:             if ($mLink == NULL) {
240:                 if (!$mMod) return FALSE; // means "..."
241:             } else {
242:                 $link = $tx->linkModule->factoryLink($mLink, $mMod, $mContent);
243:             }
244: 
245:         } elseif ($phrase === 'phrase/acronym' || $phrase === 'phrase/acronym-alt') {
246:             $mod->title = trim(Texy::unescapeHtml($mLink));
247: 
248:         } elseif ($phrase === 'phrase/quote') {
249:             $mod->cite = $tx->blockQuoteModule->citeLink($mLink);
250: 
251:         } elseif ($mLink != NULL) {
252:             $link = $tx->linkModule->factoryLink($mLink, NULL, $mContent);
253:         }
254: 
255:         return $tx->invokeAroundHandlers('phrase', $parser, array($phrase, $mContent, $mod, $link));
256:     }

294:      * Finish invocation.
295:      *
296:      * @param  TexyHandlerInvocation  handler invocation
297:      * @param  string
298:      * @param  string
299:      * @param  TexyModifier
300:      * @param  TexyLink

302:      */
303:     public function solve($invocation, $phrase, $content, $mod, $link)
304:     {
305:         $tx = $this->texy;
306: 
307:         $tag = isset($this->tags[$phrase]) ? $this->tags[$phrase] : NULL;
308: 
309:         if ($tag === 'a')
310:             $tag = $link && $this->linksAllowed ? NULL : 'span';
311: 
312:         if ($phrase === 'phrase/code')
313:             $content = $tx->protect(Texy::escapeHtml($content), Texy::CONTENT_TEXTUAL);
314: 
315:         if ($phrase === 'phrase/strong+em') {
316:             $el = TexyHtml::el($this->tags['phrase/strong']);
317:             $el->create($this->tags['phrase/em'], $content);
318:             $mod->decorate($tx, $el);
319: 
320:         } elseif ($tag) {
321:             $el = TexyHtml::el($tag)->setText($content);
322:             $mod->decorate($tx, $el);
323: 
324:             if ($tag === 'q') $el->attrs['cite'] = $mod->cite;
325:         } else {
326:             $el = $content; // trick
327:         }
328: 
329:         if ($link && $this->linksAllowed) return $tx->linkModule->solve(NULL, $link, $el);
330: 
331:         return $el;
332:     }
